Destructuring

[a, b] = [1, 2];
a === 1;
b === 2;
 
[a, b, ...c] = [1, 2, 3, 4, 5];
a === 1;
b === 2;
c === [3, 4, 5];
 
{a, b} = {a:1, b:2};
a === 1;
b === 2;

Default values :

var a, b;
[a = 7, b = 1] = [5];
 
a === 5;
b === 1;

Rest & Spread operator

var arr = [1, 2, 3];
var arr1 = [4, 5];
var arr2 = [...arr, 4];
arr.push(...arr1);

function f(a, b, ...args) {}

function f(...[a, b, c]) {}
